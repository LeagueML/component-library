import { PlatformSelector } from './platform-selector';

A component that displays a dropdown, used to select one of out all riot games platforms.

### Simple usage
```js
<PlatformSelector />
```

### Initial platform usage

```js
<PlatformSelector initialPlatform='EUW' />
```

## Preview
```js live
<PlatformSelector initialPlatform='LAN' />
```
