import { SummonerSearch } from './summoner-search';

Consists of a platform dropdown selector, a summoner name input and a search button.

### Simple usage
```js
<SummonerSearch />
```

### Initial platform usage
```js
<SummonerSearch initialPlatform='EUW' />
```

### Functional usage
```js
<SummonerSearch onSearch={() => alert('Search!')} />
```

### Preview
```js live
<SummonerSearch initialPlatform='LAN' onSearch={() => alert('Search!')} />
```
